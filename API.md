hashed\n  name?: string;\n  role: 'ADMIN' | 'MANAGER' | 'STAFF';\n  active: boolean;\n  createdAt: Date;\n  updatedAt: Date;\n}\n```\n\n### Product\n```typescript\n{\n  id: string;\n  name: string;\n  brand: string;\n  description?: string;\n  sku: string;          // unique\n  barcode?: string;\n  cost: Decimal;        // purchase price\n  price: Decimal;       // selling price\n  stock: number;\n  weight?: Decimal;\n  status: 'ACTIVE' | 'INACTIVE' | 'HIDDEN';\n  categoryId: string;\n  createdAt: Date;\n  updatedAt: Date;\n  \n  // Relations\n  category: Category;\n  images: ProductImage[];\n  inventoryHistory: InventoryHistory[];\n}\n```\n\n### Order\n```typescript\n{\n  id: string;\n  phone: string;                    // Cambodian format\n  name?: string;\n  address?: string;\n  note?: string;\n  totalAmount: Decimal;\n  paymentMethod: 'CASH' | 'ABA_QR' | 'ACLEDA' | 'WING' | 'BAKONG';\n  status: OrderStatus;              // see below\n  createdAt: Date;\n  updatedAt: Date;\n  \n  // Relations\n  items: OrderItem[];\n}\n\nenum OrderStatus {\n  PENDING,\n  CONFIRMED,\n  PACKING,\n  SHIPPING,\n  COMPLETED,\n  CANCELLED\n}\n```\n\n## Validation Schemas\n\n### Phone Validation // required\n  brand: string;          // required\n  description?: string;\n  sku: string;            // required, unique\n  barcode?: string;\n  cost: number;           // required\n  price: number;          // required, >= cost\n  stock: number;          // required\n  weight?: number;\n  categoryId: string;      // required\n  status?: 'ACTIVE' | 'INACTIVE' | 'HIDDEN';\n}\n```\n\n## Error Responses\n\nAll errors follow this format:\n```json\n{\n  \"error\": {\n    \"message\": \"Human readable error\",\n    \"status\": 400,\n    \"code\": \"ERROR_CODE\",\n    \"details\": []  //
